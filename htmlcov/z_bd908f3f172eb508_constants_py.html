4# VERSION is determined by asking the `git` executable about the current repository. 

5# This fails if not cloned, or if git is not available for some reason. 

6# VERSION should never be empty though. 

7# Note this code is duplicated in update.py for stand-alone use. 

8VERSION: str = "" 

9try: 

10 # Get the last release tag, number of commits since, and g{commit_id} as string. 

11 _VERSION_P1 = ( 

12 subprocess.check_output(["git", "describe", "--tags", "--always"]) 

13 .decode("ascii") 

14 .strip() 

15 ) 

16 # Check if any tracked files are modified for -modded version flag. 

17 _VERSION_P2 = ( 

18 subprocess.check_output( 

19 ["git", "-c", "core.fileMode=false", "status", "-suno", "--porcelain"] 

20 ) 

21 .decode("ascii") 

22 .strip() 

23 ) 

24 if _VERSION_P2: 

25 _VERSION_P2 = "-modded" 

26 else: 

27 _VERSION_P2 = "" 

28 

29 VERSION = f"{_VERSION_P1}{_VERSION_P2}" 

30 

31except (subprocess.SubprocessError, OSError, ValueError) as e: 

32 print(f"Failed setting version constant, reason: {str(e)}")
